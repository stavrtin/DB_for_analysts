Select t.user_id -- , count(tt.id_o) as n
  From 
  (
SELECT user_id, min(o_date) as mind
  FROM public.orders_20190822
  GROUP BY user_id
  ) t
  join 
    (
SELECT user_id, o_date , id_o
  FROM public.orders_20190822
--  GROUP BY user_id
  ) tt
  ON t.user_id = tt.user_id
  Where t.mind = tt.o_date
  Group BY t.user_id
  Having count(tt.id_o)  > 1;
